$$
\begin{gather}
\forall x \exists y (P(x,y)\lor \lnot Q(x,y)\rightarrow R(x,y))\\\\
\forall x \exists y (\lnot[P(x,y)\lor \lnot Q(x,y)]\lor R(x,y))\\\\
\forall x \exists y ([\lnot P(x,y)\land  Q(x,y)]\lor R(x,y))\\\\
\end{gather}

$$
